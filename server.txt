// ---------------------------------------------------------------------------
// Update channel TLS. A TLS-intercepting proxy (Zscaler) re-signs the GitHub
// certificate with its own root, which Node won't verify against the public
// store (UNABLE_TO_GET_ISSUER_CERT_LOCALLY). We let the handshake proceed,
// then inspect the certificate actually presented: if Node already trusts the
// chain we accept normally; if not, we accept ONLY when the failure is an
// unknown-issuer error AND the chain is a Zscaler interception cert. Anything
// else (expired, hostname mismatch, unknown self-signed) is still rejected.
// Applied files remain integrity-checked by SHA-256 against checksums.txt.
// ---------------------------------------------------------------------------
const _ISSUER_ERRS = /UNABLE_TO_GET_ISSUER_CERT_LOCALLY|SELF_SIGNED_CERT_IN_CHAIN|UNABLE_TO_VERIFY_LEAF_SIGNATURE|DEPTH_ZERO_SELF_SIGNED_CERT/
function chainIsZscaler(socket) {
  try {
    let c = socket.getPeerCertificate(true)
    const seen = new Set()
    while (c && c.fingerprint256 && !seen.has(c.fingerprint256)) {
      seen.add(c.fingerprint256)
      const fields = [c.issuer && c.issuer.O, c.issuer && c.issuer.CN,
                      c.subject && c.subject.O, c.subject && c.subject.CN].join(' ')
      if (/zscaler/i.test(fields)) return true
      if (c.issuerCertificate && c.issuerCertificate !== c) c = c.issuerCertificate
      else break
    }
  } catch {}
  return false
}
// True if a socket's cert is untrusted AND not a Zscaler-intercepted chain.
// Lets PlatformAI calls succeed behind Zscaler while still rejecting genuinely
// untrusted certs. Used by the main proxy request handlers below.
function tlsGateFails(sock) {
  if (!sock || sock.authorized !== false) return false
  const ae = String(sock.authorizationError || '')
  return !(_ISSUER_ERRS.test(ae) && chainIsZscaler(sock))
}

// ---------------------------------------------------------------------------
// Binary embed cache — Float16, 2 bytes per float, keyed by SHA1 hash
// ---------------------------------------------------------------------------
const EMBED_CACHE_FILE = path.join(LCL_DIR, 'embed_cache.bin')
const MAGIC = 0x4C434C43
const embedCache = {}

function encodeFloat16(val) {
  if (isNaN(val))  return 0x7e00
  if (val === 0)   return 0
  const f32  = new Float32Array([val])
  const u32  = new Uint32Array(f32.buffer)[0]
  const sign = (u32 >> 31) & 0x1
  let   exp  = ((u32 >> 23) & 0xff) - 127 + 15
  let   mant = (u32 >> 13) & 0x3ff
  if (exp <= 0)  return (sign << 15)
  if (exp >= 31) return (sign << 15) | (0x1f << 10)
  return (sign << 15) | (exp << 10) | mant
}

function decodeFloat16(h) {
  const sign = (h >> 15) & 0x1
  const exp  = (h >> 10) & 0x1f
  const mant =  h        & 0x3ff
  let val
  if      (exp === 0)  val = mant / 1024 * Math.pow(2, -14)
  else if (exp === 31) val = mant ? NaN : Infinity
  else                 val = (1 + mant / 1024) * Math.pow(2, exp - 15)
  return sign ? -val : val
}

function loadEmbedCache() {
  try {
    if (!fs.existsSync(EMBED_CACHE_FILE)) return {}
    const buf = fs.readFileSync(EMBED_CACHE_FILE)
    if (buf.length < 8) return {}
    const magic = buf.readUInt32BE(0)
    if (magic !== MAGIC) {
      elog('cache: unrecognised magic — delete embed_cache.bin to silence this warning')
      return {}
    }
    const count  = buf.readUInt32LE(4)
    const result = {}
    let   offset = 8
    for (let i = 0; i < count; i++) {
      if (offset + 4 > buf.length) break
      const hashLen = buf.readUInt32LE(offset); offset += 4
      if (offset + hashLen > buf.length) break
      const hash = buf.toString('utf8', offset, offset + hashLen); offset += hashLen
      if (offset + 4 > buf.length) break
      const dims    = buf.readUInt32LE(offset); offset += 4
      const byteLen = dims * 2
      if (offset + byteLen > buf.length) break
      const f32 = new Float32Array(dims)
      for (let j = 0; j < dims; j++) {
        f32[j] = decodeFloat16(buf.readUInt16LE(offset + j * 2))
      }
      result[hash] = f32
      offset += byteLen
    }
    elog('cache: loaded ' + Object.keys(result).length + ' entries (float16)')
    return result
  } catch (e) {
    elog('cache load error: ' + e.message)
    return {}
  }
}

function saveEmbedCache() {
  try {
    const entries = Object.entries(embedCache)
    if (entries.length === 0) {
      const buf = Buffer.allocUnsafe(8)
      buf.writeUInt32BE(MAGIC, 0)
      buf.writeUInt32LE(0, 4)
      fs.writeFileSync(EMBED_CACHE_FILE, buf)
      return
    }
    let totalSize = 8
    for (const [hash, vec] of entries) {
      totalSize += 4 + Buffer.byteLength(hash, 'utf8') + 4 + vec.length * 2
    }
    const buf    = Buffer.allocUnsafe(totalSize)
    let   offset = 0
    buf.writeUInt32BE(MAGIC, offset); offset += 4
    buf.writeUInt32LE(entries.length, offset); offset += 4
    for (const [hash, vec] of entries) {
      const hashBuf = Buffer.from(hash, 'utf8')
      buf.writeUInt32LE(hashBuf.length, offset); offset += 4
      hashBuf.copy(buf, offset); offset += hashBuf.length
      buf.writeUInt32LE(vec.length, offset); offset += 4
      for (let j = 0; j < vec.length; j++) {
        buf.writeUInt16LE(encodeFloat16(vec[j]), offset); offset += 2
      }
    }
    fs.writeFileSync(EMBED_CACHE_FILE, buf)
  } catch (e) {
    elog('cache save error: ' + e.message)
  }
}

function hashText(s) {
  return crypto.createHash('sha1').update(s).digest('base64url').slice(0, 16)
}

// Only accept requests whose Host header points at loopback. Blocks DNS-rebinding
// attacks and stops any accidental exposure if the bind address is ever widened.
function isLocalHost(req) {
  const h = (req.headers.host || '').split(':')[0].toLowerCase()
  return h === 'localhost' || h === '127.0.0.1' || h === '[::1]' || h === '::1'
}

function readBody(req, cb) {
  let body = ''

  req.on('data', chunk => {
    body += chunk
    if (body.length > 10 * 1024 * 1024) {
      cb(new Error('Body too large'))
      req.destroy()
    }
  })

  req.on('end', () => {
    if (!body.trim()) return cb(new Error('Empty body'))
    try {
      cb(null, JSON.parse(body))
    } catch {
      cb(new Error('Bad JSON'))
    }
  })

  req.on('error', err => cb(err))
}

function maskSecret(v) {
  if (!v) return '(not set)'
  if (v.length <= 10) return '[set]'
  return v.slice(0, 4) + '...' + v.slice(-4)
}

function previewText(v, max = 4000) {
  if (!v) return ''
  return v.length > max ? v.slice(0, max) + ' ...[truncated]' : v
}

// ---------------------------------------------------------------------------
// Rate-limit header parser
// PlatformAI returns x-ratelimit-api_key-* headers on every response. We parse
// them once and pass the result alongside the body so callers can:
//   - log per-call usage,
//   - surface remaining quota to the client,
//   - pre-emptively slow down before hitting a 429.
// Headers seen in the wild:
//   x-ratelimit-api_key-remaining-requests
//   x-ratelimit-api_key-limit-requests
//   x-ratelimit-api_key-remaining-tokens
//   x-ratelimit-api_key-limit-tokens
// ---------------------------------------------------------------------------
function extractRateLimit(headers) {
  if (!headers) return null
  const h = (k) => headers[k.toLowerCase()] ?? headers[k]
  const reqRem = h('x-ratelimit-api_key-remaining-requests')
  const reqLim = h('x-ratelimit-api_key-limit-requests')
  const tokRem = h('x-ratelimit-api_key-remaining-tokens')
  const tokLim = h('x-ratelimit-api_key-limit-tokens')
  if (reqRem == null && tokRem == null) return null
  const num = (v) => (v == null ? null : Number(v))
  return {
    reqRemaining: num(reqRem),
    reqLimit:     num(reqLim),
    tokRemaining: num(tokRem),
    tokLimit:     num(tokLim),
  }
}

// Track the most recent rate-limit snapshot per (apiKey prefix). Used by the
// embed batch loop to adapt its pacing without needing the previous response
// in scope. Keyed on the first 6 chars of the masked key so we don't keep the
// full secret in memory.
const rateLimitSnapshots = new Map()
function snapshotKey(apiKey) {
  if (!apiKey) return '_'
  return apiKey.slice(0, 6)
}
function getLastRateLimit(apiKey) {
  return rateLimitSnapshots.get(snapshotKey(apiKey)) || null
}
function setLastRateLimit(apiKey, rl) {
  if (rl) rateLimitSnapshots.set(snapshotKey(apiKey), { ...rl, ts: Date.now() })
}

  timeout: 60_000
})

// Change 2: TLS/CERT RETRY FIX — destroy socket on cert/TLS errors
// ---------------------------------------------------------------------------
function callGccJsonOnce(pathName, apiKey, payload, cb) {
  const bodyStr = JSON.stringify(payload)

  const opts = {
    hostname: API_HOST,
    port: 443,
    path: pathName,
    method: 'POST',
    agent: upstreamAgent,
    headers: {
      'x-api-key': apiKey,
      'Content-Type': 'application/json',
      'Content-Length': Buffer.byteLength(bodyStr),
      'Accept': '*/*',
      'User-Agent': 'curl/8.0.0',
      'Connection': 'keep-alive'
    }
  }

  console.log('')
  console.log('========== OUTBOUND REQUEST ==========')
  console.log('[api] time       =', new Date().toISOString())
  console.log('[api] hostname   =', opts.hostname)
  console.log('[api] path       =', opts.path)
  console.log('[api] auth       =', maskSecret(apiKey))
  console.log('[api] payload    =', previewText(bodyStr))
  console.log('======================================')
  console.log('')

  // Guard against double-invocation: on timeout we'll destroy() the request,
  // which can race with 'end' or 'error' firing. done() ensures cb runs once
  // and always clears the total-budget timer.
  let settled = false
  let totalBudgetTimer = null
  const done = (err, result) => {
    if (settled) return
    settled = true
    if (totalBudgetTimer) clearTimeout(totalBudgetTimer)
    cb(err, result)
  }

  const req = https.request(opts, upstreamRes => {
    if (tlsGateFails(upstreamRes.socket)) {
      const ae = String(upstreamRes.socket.authorizationError || 'TLS not trusted')
      upstreamRes.resume(); try { upstreamRes.socket.destroy() } catch {}
      console.log('[api] TLS rejected =', ae)
      return done(Object.assign(new Error('TLS certificate not trusted: ' + ae), { code: ae }))
    }
    let body = ''

    console.log('[api] response status  =', upstreamRes.statusCode)
    console.log('[api] response headers =', upstreamRes.headers)

    upstreamRes.on('data', chunk => {
      body += chunk
      console.log('[api] chunk bytes =', Buffer.byteLength(chunk))
    })

    upstreamRes.on('end', () => {
      console.log('[api] response body length =', body.length)
      console.log('[api] response body =', previewText(body))
      const rl = extractRateLimit(upstreamRes.headers)
      if (rl) {
        // One-line summary so console scans cleanly without dumping full headers.
        const fmt = (rem, lim) => (rem == null ? '?' : rem) + '/' + (lim == null ? '?' : lim)
        console.log('[api] rate-limit  req=' + fmt(rl.reqRemaining, rl.reqLimit) +
                    '  tok=' + fmt(rl.tokRemaining, rl.tokLimit))
        setLastRateLimit(apiKey, rl)
      }
      done(null, {
        statusCode: upstreamRes.statusCode || 500,
        headers: upstreamRes.headers,
        body,
        rateLimit: rl
      })
    })

    upstreamRes.on('error', err => {
      console.log('[api] response stream error =', err.message)
      done(err)
    })
  })

  // Inactivity timeout: no bytes moved in 90s -> treat as dead and retry.
  // Generous because Claude Opus 4.6 on a long prompt may take 60s+ to emit
  // its first byte when stream=false.
  req.setTimeout(90_000, () => {
    console.log('[api] inactivity timeout (90s)')
    req.destroy(new Error('Upstream inactivity timeout'))
  })

  // Hard overall budget: inactivity alone isn't enough - a trickling upstream
  // can keep a socket alive for minutes. 120s is the worst-case cap per attempt.
  totalBudgetTimer = setTimeout(() => {
    console.log('[api] total-budget exceeded (120s), aborting')
    req.destroy(new Error('Upstream total-budget exceeded'))
  }, 120_000)

  req.on('socket', socket => {
    console.log('[api] socket assigned')
    socket.on('secureConnect', () => {
      console.log('[api] TLS connected')
      console.log('[api] authorized =', socket.authorized)
      console.log('[api] authorizationError =', socket.authorizationError || '(none)')
    })
    // Change 2: destroy socket on cert/TLS errors to force fresh connection on retry
    socket.on('error', err => {
      console.log('[api] socket error =', err.message)
      if (/certificate|CERT|TLS|SSL|issuer/i.test(err.message)) {
        socket.destroy()
      }
    })
  })

  req.on('finish', () => {
    console.log('[api] request body sent')
  })

  req.on('error', err => {
    console.log('[api] request error =', err.message)
    done(err)
  })

  req.write(bodyStr)
  req.end()
}

// Retry wrapper: transparently re-issue the call on transient upstream failures
// (5xx, empty body, non-JSON body, socket errors, TLS/cert errors).
// Uses exponential backoff with jitter. `maxAttempts` includes the first try.
// ---------------------------------------------------------------------------
// Change 2 (continued): cert/TLS errors treated as transient
// Change 3: baseDelay reduced from 400 to 150ms
// ---------------------------------------------------------------------------
function callGccJson(pathName, apiKey, payload, cb, opts = {}) {
  const maxAttempts = opts.maxAttempts || 4
  const baseDelay   = opts.baseDelay   || 150   // ms  (was 400)
  const maxDelay    = opts.maxDelay    || 4000  // ms

  let attempt = 0

  const tryOnce = () => {
    attempt++
    callGccJsonOnce(pathName, apiKey, payload, (err, upstream) => {
      const transient = (() => {
        if (err && /certificate|CERT|TLS|SSL|issuer/i.test(err.message)) return true
        if (err) return true
        if (!upstream) return true
        const s = upstream.statusCode
        if (s === 502 || s === 503 || s === 504 || s === 429) return true
        const b = upstream.body
        if (!b || !b.trim()) return true
        // Non-JSON response from what should be a JSON endpoint.
        try { JSON.parse(b); return false } catch { return true }
      })()

      if (!transient || attempt >= maxAttempts) return cb(err, upstream)

      const delay = Math.min(maxDelay, baseDelay * 2 ** (attempt - 1))
      const jitter = Math.floor(Math.random() * 200)
      console.log(`[api] transient failure (attempt ${attempt}/${maxAttempts}) - retrying in ${delay + jitter}ms`)
      setTimeout(tryOnce, delay + jitter)
    })
  }

  tryOnce()
}

// ---------------------------------------------------------------------------
// Streaming proxy. When the client sends payload.stream === true we open the
// upstream and pipe its SSE bytes straight back — no buffering, no JSON parse.
// The 90s inactivity timeout effectively becomes a non-issue because tokens
// trickle in continuously. Auto-title and any other non-streamed callers
// still go through the buffered callGccJson path.
// ---------------------------------------------------------------------------
function callGccStreaming(pathName, apiKey, payload, res, timeoutMs) {
  const bodyStr = JSON.stringify(payload)
  const opts = {
    hostname: API_HOST, port: 443, path: pathName, method: 'POST',
    agent: upstreamAgent,
    headers: {
      'x-api-key': apiKey,
      'Content-Type':  'application/json',
      'Content-Length': Buffer.byteLength(bodyStr),
      'Accept':        'text/event-stream',
      'User-Agent':    'curl/8.0.0',
      'Connection':    'keep-alive'
    }
  }

  console.log('')
  console.log('========== STREAMING OUTBOUND ==========')
  console.log('[stream] time     =', new Date().toISOString())
  console.log('[stream] path     =', pathName)
  console.log('[stream] auth     =', maskSecret(apiKey))
  console.log('[stream] payload  =', previewText(bodyStr))
  console.log('========================================')

  const upReq = https.request(opts, upRes => {
    if (tlsGateFails(upRes.socket)) {
      const ae = String(upRes.socket.authorizationError || 'TLS not trusted')
      upRes.resume(); try { upRes.socket.destroy() } catch {}
      console.log('[stream] TLS rejected =', ae)
      if (!res.headersSent) res.writeHead(502, { 'Content-Type': 'application/json; charset=utf-8' })
      res.end(JSON.stringify({ error: 'TLS certificate not trusted (' + ae + ')' }))
      return
    }
    console.log('[stream] response status  =', upRes.statusCode)
    const rl = extractRateLimit(upRes.headers)
    if (rl) {
      setLastRateLimit(apiKey, rl)
      const fmt = (rem, lim) => (rem == null ? '?' : rem) + '/' + (lim == null ? '?' : lim)
      console.log('[stream] rate-limit  req=' + fmt(rl.reqRemaining, rl.reqLimit) +
                  '  tok=' + fmt(rl.tokRemaining, rl.tokLimit))
    }

    // Non-200: drain the body and surface it as a single JSON error event.
    if (upRes.statusCode !== 200) {
      let body = ''
      upRes.on('data', c => body += c)
      upRes.on('end', () => {
        console.log('[stream] non-200 body =', previewText(body))
        if (!res.headersSent) {
          res.writeHead(upRes.statusCode, { 'Content-Type': 'application/json; charset=utf-8' })
        }
        let parsed
        try { parsed = JSON.parse(body) }
        catch { parsed = { error: { message: 'Upstream returned HTTP ' + upRes.statusCode + ' (non-JSON response)' }, status: upRes.statusCode } }
        if (rl) parsed._rateLimit = rl
        res.end(JSON.stringify(parsed))
      })
      return
    }

    // 200 OK: pipe SSE bytes straight to the client.
    res.writeHead(200, {
      'Content-Type':  'text/event-stream; charset=utf-8',
      'Cache-Control': 'no-cache',
      'Connection':    'keep-alive'
    })
    upRes.on('data', chunk => res.write(chunk))
    upRes.on('end', () => { console.log('[stream] upstream end'); try { res.end() } catch {} })
    upRes.on('error', err => {
      console.log('[stream] upstream stream error =', err.message)
      try { res.end() } catch {}
    })
  })

  // Inactivity timeout (no NEW bytes for this long) escalates per client retry
  // attempt (10s -> 20s -> 60s), passed in via streamTimeoutMs. The socket timer
  // resets on every token, so it only fires on a genuine stall. Clamped 5s-120s.
  const inactivityMs = Math.min(120000, Math.max(5000, Number(timeoutMs) || 60000))
  upReq.setTimeout(inactivityMs, () => {
    console.log('[stream] inactivity timeout (' + Math.round(inactivityMs / 1000) + 's)')
    upReq.destroy(new Error('Upstream inactivity timeout'))
  })

  upReq.on('error', err => {
    console.log('[stream] request error =', err.message)
    if (!res.headersSent) {
      res.writeHead(502, { 'Content-Type': 'application/json; charset=utf-8' })
      res.end(JSON.stringify({ error: err.message, kind: 'upstream_unreachable' }))
    } else {
      try {
        res.write('data: ' + JSON.stringify({ error: err.message }) + '\n\n')
        res.end()
      } catch {}
    }
  })

  // If the browser aborts (user clicked Stop), kill the upstream so we don't
  // keep burning tokens.
  res.on('close', () => {
    if (!upReq.destroyed) {
      console.log('[stream] client disconnected, aborting upstream')
      upReq.destroy()
    }
  })

  upReq.write(bodyStr)
  upReq.end()
}

// ---------------------------------------------------------------------------
function handleChat(req, res) {
  readBody(req, (err, body) => {
    if (err) return json(res, { error: err.message }, 400)

    const settings = appData.settings || {}
    const apiKey  = settings.apiKey  || body.apiKey
    const modelId = settings.modelId || body.modelId
    const { payload } = body

    if (!apiKey || !modelId || !payload) {
      return json(res, { error: 'Missing params' }, 400)
    }

    const finalPayload = { ...payload, model: modelId }

    // Streaming path for live token-by-token rendering
    if (finalPayload.stream === true) {
      return callGccStreaming(CHAT_PATH, apiKey, finalPayload, res, body.streamTimeoutMs)
    }

    // Buffered path for short utility calls (auto-title, embed test, etc.)
    callGccJson(CHAT_PATH, apiKey, finalPayload, (err, upstream) => {
      if (err) {
        // 504 = Gateway Timeout. Correct signal for "we couldn't reach upstream";
        // lets clients distinguish network failure from a real server bug.
        const isTimeout = /timeout|budget/i.test(err.message)
        return json(res, { error: err.message, kind: 'upstream_unreachable' },
                    isTimeout ? 504 : 502)
      }

      const { statusCode, headers, body: respBody } = upstream

      if (!respBody || !respBody.trim()) {
        return json(res, {
          error: 'Empty response from upstream',
          status: statusCode,
          headers
        }, statusCode >= 400 ? statusCode : 502)
      }

      try {
        const parsed = JSON.parse(respBody)
        if (upstream.rateLimit) parsed._rateLimit = upstream.rateLimit
        return json(res, parsed, statusCode)
      } catch {
        return json(res, {
          error: 'Upstream returned non-JSON response',
          status: statusCode,
          headers,
          raw: respBody,
          _rateLimit: upstream.rateLimit || null
        }, statusCode >= 400 ? statusCode : 502)
      }
    })
  })
}

// ---------------------------------------------------------------------------
// Batch embed — helpers
// ---------------------------------------------------------------------------
// Per-model embedding limits — batch = max texts per request, maxChars = per-text
// cap. Cohere Embed v3: 96 texts, ~512 tokens (~2000 chars). OpenAI
// text-embedding-3: 2048 inputs, ~8191 tokens (~30000 chars). Unknown models fall
// back to the conservative Cohere caps so we never exceed a provider limit.
const EMBED_LIMITS = {
  'cohere.embed-english-v3':      { batch: 96,   maxChars: 2000 },   // 512-token inputs
  'cohere.embed-multilingual-v3': { batch: 96,   maxChars: 2000 },
  'cohere.embed-v4:0':            { batch: 96,   maxChars: 8000 },   // larger context, array cap still 96
  'text-embedding-3-small':       { batch: 2048, maxChars: 30000 },  // 8191-token inputs
  'text-embedding-3-large':       { batch: 2048, maxChars: 30000 },
  'text-embedding-ada-002':       { batch: 2048, maxChars: 30000 },
  // gemini-embedding-001: batching IS supported (batchEmbedContents); inputs cap
  // at 8192 tokens. 96 sits safely under Gemini's ~100 / Vertex's ~250 documented
  // per-request caps; lower it only if the gateway returns a 400.
  'gemini-embedding-001':         { batch: 96,   maxChars: 8000 }
}
function embedLimitsFor(modelId) {
  if (EMBED_LIMITS[modelId]) return EMBED_LIMITS[modelId]
  const m = String(modelId || '').toLowerCase()
  if (m.includes('text-embedding')) return { batch: 2048, maxChars: 30000 }
  if (m.includes('cohere') || m.includes('embed-')) return { batch: 96, maxChars: 2000 }
  return { batch: 96, maxChars: 2000 }   // conservative default
}

function callGccJsonAsync(pathName, apiKey, payload) {
  return new Promise((resolve, reject) => {
    callGccJson(pathName, apiKey, payload, (err, upstream) => {
      if (err) return reject(err)
      resolve(upstream)
    })
  })
}

async function callStandardBatchEmbed(apiKey, modelId, texts) {
  elog('batch embed | ' + texts.length + ' chunks')
  const upstream = await callGccJsonAsync(EMBED_PATH, apiKey, { model: modelId, input: texts })

  if (!upstream.body || !upstream.body.trim()) {
    throw new Error('Empty response from embeddings (status ' + upstream.statusCode + ')')
  }

  let data
  try { data = JSON.parse(upstream.body) }
  catch { throw new Error('Non-JSON response: ' + upstream.body.slice(0, 200)) }

  if (upstream.statusCode >= 400) {
    throw new Error('Embeddings error ' + upstream.statusCode + ': ' + JSON.stringify(data))
  }

  const sorted = (data?.data || []).sort((a, b) => a.index - b.index)
  const embeddings = sorted.map(e => e.embedding)

  if (embeddings.length !== texts.length) {
    throw new Error('Embedding count mismatch: got ' + embeddings.length + ', expected ' + texts.length)
  }
  return embeddings
}

const sleep = ms => new Promise(r => setTimeout(r, ms))

// Pace embed batches against the per-minute rate-limit window. The PlatformAI
// key allows only a small number of requests/tokens per ~60s; bursting several
// batches can trip a 429 the short retry can't outlast. Before a batch, if the
// last snapshot shows we're near the cap, wait for the window to reset and call
// notify(secsLeft) each tick so the client can show a clear countdown instead
// of looking frozen. tokenEst = rough token cost of the upcoming batch.
async function waitForEmbedHeadroom(apiKey, tokenEst, notify) {
  const WINDOW_MS = 60000
  const rl = getLastRateLimit(apiKey)
  if (!rl) return
  const reqLow = rl.reqRemaining != null && rl.reqRemaining <= 2
  const tokLow = rl.tokRemaining != null && rl.tokRemaining <= (tokenEst || 25000)
  if (!reqLow && !tokLow) return
  const age = Date.now() - (rl.ts || 0)
  let remain = Math.max(2000, WINDOW_MS - age)
  elog('pacing | req=' + rl.reqRemaining + ', tok=' + rl.tokRemaining +
       ' near cap; waiting ~' + Math.round(remain / 1000) + 's for rate-limit window reset')
  while (remain > 0) {
    if (notify) { try { notify(Math.ceil(remain / 1000)) } catch {} }
    const step = Math.min(3000, remain)
    await sleep(step)
    remain -= step
  }
  rateLimitSnapshots.delete(snapshotKey(apiKey))   // window reset - re-measure next call
}

async function handleEmbedBatch(req, res) {
  readBody(req, async (err, body) => {
    if (err) return json(res, { error: err.message }, 400)

    const settings = appData.settings || {}
    const apiKey   = settings.embedApiKey  || body.apiKey
    const modelId  = settings.embedModelId || body.modelId
    const lim = embedLimitsFor(modelId)
    const { inputs } = body

    if (!apiKey || !modelId || !inputs?.length) {
      return json(res, { error: 'Missing apiKey, modelId, or inputs' }, 400)
    }

    const allHashes = inputs.map(t => hashText(modelId + ':' + t))

    // ── Cache check ──
    const results = new Array(inputs.length).fill(null)
    const missIdx = []
    for (let i = 0; i < inputs.length; i++) {
      const key = hashText(modelId + ':' + inputs[i])
      if (embedCache[key]) {
        results[i] = embedCache[key]
      } else {
        missIdx.push(i)
      }
    }

    elog('cache check | total ' + inputs.length + ' | hits ' + (inputs.length - missIdx.length) + ' | misses ' + missIdx.length)

    // All cached — return plain JSON
    if (!missIdx.length) {
      return json(res, { embeddings: results.map(v => Array.from(v)), hashes: allHashes })
    }

    // SSE for partial / full miss
    res.writeHead(200, {
      'Content-Type':      'text/event-stream; charset=utf-8',
      'Cache-Control':     'no-cache',
      'Connection':        'keep-alive',
      'X-Accel-Buffering': 'no'
    })
    const send = obj => res.write('data: ' + JSON.stringify(obj) + '\n\n')

    try {
      let chunksEmbedded = 0
      const totalMisses  = missIdx.length
      const totalBatches = Math.ceil(totalMisses / lim.batch)

      for (let b = 0; b < missIdx.length; b += lim.batch) {
        const batchOrigIdx = missIdx.slice(b, b + lim.batch)
        const batchTexts   = batchOrigIdx.map(i => inputs[i].slice(0, lim.maxChars))
        const batchNum     = Math.floor(b / lim.batch) + 1

        elog('batch ' + batchNum + '/' + totalBatches + ' | ' + batchTexts.length + ' chunks')

        await waitForEmbedHeadroom(apiKey, batchTexts.length * 250, secsLeft =>
          send({ type: 'pacing', waitSec: secsLeft, done: chunksEmbedded, total: totalMisses,
                 batchDone: batchNum, batchTotal: totalBatches }))

        const embeddings = await callStandardBatchEmbed(apiKey, modelId, batchTexts)

        for (let i = 0; i < embeddings.length; i++) {
          const origIdx  = batchOrigIdx[i]
          const cacheKey = hashText(modelId + ':' + inputs[origIdx])
          const f32      = new Float32Array(embeddings[i])
          results[origIdx]     = f32
          embedCache[cacheKey] = f32
        }

        chunksEmbedded += batchTexts.length
        saveEmbedCache()

        send({
          type:       'progress',
          done:       chunksEmbedded,
          total:      totalMisses,
          batchDone:  batchNum,
          batchTotal: totalBatches
        })
      }

      send({ type: 'done', embeddings: results.map(v => Array.from(v)), hashes: allHashes })
      res.end()
    } catch (e) {
      elog('batch error: ' + e.message)
      send({ type: 'error', message: e.message })
      res.end()
    }
  })
}

function handleEmbedLookup(req, res) {
  readBody(req, (err, body) => {
    if (err) return json(res, { error: err.message }, 400)

    const { hashes } = body
    if (!Array.isArray(hashes) || !hashes.length || hashes.length > 1500) {
      return json(res, { error: 'Invalid hashes array' }, 400)
    }

    const vectors = hashes.map(h => {
      const v = embedCache[h]
      return v ? Array.from(v) : null
    })

    const hits = vectors.filter(v => v !== null).length
    elog('lookup | requested ' + hashes.length + ' | hits ' + hits + ' | misses ' + (hashes.length - hits))

    json(res, { vectors })
  })
}

function handleEmbedEvict(req, res) {
  readBody(req, (err, body) => {
    if (err) return json(res, { error: err.message }, 400)

    const { hashes } = body
    if (!Array.isArray(hashes) || !hashes.length || hashes.length > 1500) {
      return json(res, { error: 'Invalid hashes array' }, 400)
    }

    let removed = 0
    for (const h of hashes) {
      if (embedCache[h]) { delete embedCache[h]; removed++ }
    }

    elog('evict | requested ' + hashes.length + ' | removed ' + removed)
    if (removed > 0) saveEmbedCache()

    json(res, { ok: true, removed })
  })
}

// ---------------------------------------------------------------------------
// Embed-cache garbage collection. The on-disk vector cache (embed_cache.bin)
// can outlive the docs that produced it (removed docs, old orphans). prune
// keeps only vectors still referenced by a chunk in some saved doc, and never
// drops a vector another doc still shares. Run on doc removal + at startup.
// ---------------------------------------------------------------------------
function referencedHashes() {
  const keep = new Set()
  const chats = (appData && appData.chats) || {}
  for (const c of Object.values(chats)) {
    if (!Array.isArray(c.docs)) continue
    for (const d of c.docs) {
      if (!Array.isArray(d.chunks)) continue
      for (const ch of d.chunks) if (typeof ch.embHash === 'string') keep.add(ch.embHash)
    }
  }
  return keep
}

function pruneEmbedCache() {
  const keep = referencedHashes()
  let removed = 0
  for (const h of Object.keys(embedCache)) {
    if (!keep.has(h)) { delete embedCache[h]; removed++ }
  }
  if (removed) saveEmbedCache()
  elog('cache prune | kept ' + keep.size + ' | removed ' + removed)
  return removed
}

function handleEmbedGc(req, res) {
  json(res, { ok: true, removed: pruneEmbedCache() })
}

// Change 4: handleSaveData — do NOT strip settings from the saved body.
// Preserve existing settings if the incoming body doesn't include them.
// ---------------------------------------------------------------------------
function handleSaveData(req, res) {
  readBody(req, (err, body) => {
    if (err) return json(res, { error: err.message }, 400)
    // Preserve existing settings if the incoming body doesn't include them
    if (!body.settings && appData.settings) {
      body.settings = appData.settings
    }
    if (body.updateChannel === undefined && appData.updateChannel !== undefined) {
      body.updateChannel = appData.updateChannel
    }
    // Strip full embedding vectors — only embHash is needed on disk
    if (body && body.chats) {
      for (const chat of Object.values(body.chats)) {
        if (!Array.isArray(chat.docs)) continue
        for (const doc of chat.docs) {
          if (!Array.isArray(doc.chunks)) continue
          for (const chunk of doc.chunks) {
            delete chunk.embedding
          }
        }
      }
    }
    appData = body
    saveData(appData)
    json(res, { ok: true })
  })
}

function handleSkillsList(req, res) {
  let entries
  try {
    entries = fs.readdirSync(SKILLS_DIR)
  } catch (e) {
    return json(res, { error: 'Cannot read skills directory: ' + e.message }, 500)
  }

  const skills = []
  for (const name of entries) {
    if (!name.endsWith('.md')) continue
    const id = name.slice(0, -3)
    if (!isValidSlug(id)) {
      console.warn('[skills] ignoring file with invalid slug:', name)
      continue
    }
    const fp = path.join(SKILLS_DIR, name)
    let stat
    try { stat = fs.statSync(fp) } catch { continue }
    if (!stat.isFile()) continue
    let title = null
    if (stat.size <= SKILL_MAX_BYTES) {
      try {
        const body = fs.readFileSync(fp, 'utf8')
        title = readSkillTitleFromBody(body)
      } catch {}
    }
    skills.push({
      id,
      title: title || id,
      bytes: stat.size,
      mtime: stat.mtimeMs
    })
  }
  skills.sort((a, b) => a.id.localeCompare(b.id))
  json(res, { skills })
}

function handleSkillRead(req, res, id) {
  if (!isValidSlug(id)) return json(res, { error: 'Invalid skill id' }, 400)
  const fp = skillPath(id)
  let stat
  try { stat = fs.statSync(fp) }
  catch { return json(res, { error: 'Not found' }, 404) }
  if (!stat.isFile()) return json(res, { error: 'Not found' }, 404)
  if (stat.size > SKILL_MAX_BYTES) {
    return json(res, { error: 'Skill exceeds size cap (' + SKILL_MAX_BYTES + ' bytes)' }, 413)
  }
  let body
  try { body = fs.readFileSync(fp, 'utf8') }
  catch (e) { return json(res, { error: 'Read failed: ' + e.message }, 500) }
  json(res, { id, title: readSkillTitleFromBody(body) || id, body })
}

function handleSkillWrite(req, res, id) {
  if (!isValidSlug(id)) return json(res, { error: 'Invalid skill id' }, 400)
  readBody(req, (err, body) => {
    if (err) return json(res, { error: err.message }, 400)
    if (!body || typeof body.body !== 'string') {
      return json(res, { error: 'Expected JSON { body: string }' }, 400)
    }
    if (body.body.indexOf('\0') !== -1) {
      return json(res, { error: 'Body contains NUL byte' }, 400)
    }
    const bytes = Buffer.byteLength(body.body, 'utf8')
    if (bytes > SKILL_MAX_BYTES) {
      return json(res, { error: 'Skill exceeds size cap (' + SKILL_MAX_BYTES + ' bytes)' }, 413)
    }
    const fp = skillPath(id)
    try { fs.writeFileSync(fp, body.body, 'utf8') }
    catch (e) { return json(res, { error: 'Write failed: ' + e.message }, 500) }
    let stat
    try { stat = fs.statSync(fp) } catch { stat = { size: bytes, mtimeMs: Date.now() } }
    json(res, {
      id,
      title: readSkillTitleFromBody(body.body) || id,
      bytes: stat.size,
      mtime: stat.mtimeMs
    })
  })
}

const MIME = {
  '.html': 'text/html; charset=utf-8',
  '.js': 'application/javascript; charset=utf-8',
  '.css': 'text/css; charset=utf-8',
  '.json': 'application/json; charset=utf-8',
  '.txt': 'text/plain; charset=utf-8'
}

function serveStatic(req, res) {
  const reqPath   = req.url === '/' ? '/index.html' : req.url
  const cleanPath = path.normalize(reqPath).replace(/^(\.\.[/\\])+/, '')
  const fp        = path.join(LCL_DIR, cleanPath)

  if (!fp.startsWith(path.resolve(LCL_DIR) + path.sep) && fp !== path.resolve(LCL_DIR)) {
    res.writeHead(403, { 'Content-Type': 'text/plain; charset=utf-8' })
    res.end('Forbidden')
    return
  }

  fs.readFile(fp, (err, data) => {
    if (err) {
      res.writeHead(404, { 'Content-Type': 'text/plain; charset=utf-8' })
      res.end('Not found')
      return
    }

    // Always revalidate: the browser sends If-None-Match and gets a tiny 304 when
    // the file is unchanged, or the fresh bytes when it changed — so rebuilds show
    // up on a normal reload without re-downloading every file every time.
    const etag = '"' + crypto.createHash('sha1').update(data).digest('hex').slice(0, 16) + '"'
    if (req.headers['if-none-match'] === etag) {
      res.writeHead(304, { 'ETag': etag, 'Cache-Control': 'no-cache' })
      res.end()
      return
    }
    res.writeHead(200, {
      'Content-Type': MIME[path.extname(fp).toLowerCase()] || 'application/octet-stream',
      'ETag': etag,
      'Cache-Control': 'no-cache'
    })
    res.end(data)
  })
}

// ---------------------------------------------------------------------------
// Auto-update engine. Checks GitHub releases/latest; on consent, downloads the
// files at the release tag (raw URLs), verifies each against the published
// checksums.txt (sha256) into a .tmp, and only then atomically renames over the
// live file. index.html applies on browser refresh; server.txt needs a Node
// restart. Nothing outside LCL_DIR is ever written.
// ---------------------------------------------------------------------------
function cmpVer(a, b) {
  const norm = v => String(v || '').replace(/^v/i, '').trim()
  const re = /\d+|[a-zA-Z]+/g
  const ta = norm(a).match(re) || [], tb = norm(b).match(re) || []
  const n = Math.max(ta.length, tb.length)
  for (let i = 0; i < n; i++) {
    const x = ta[i], y = tb[i]
    if (x === undefined) return -1
    if (y === undefined) return 1
    const xn = /^\d+$/.test(x), yn = /^\d+$/.test(y)
    if (xn && yn) { const d = parseInt(x, 10) - parseInt(y, 10); if (d) return d > 0 ? 1 : -1 }
    else if (x !== y) { return x > y ? 1 : -1 }
  }
  return 0
}

function httpsGetBuf(url, headers = {}) {
  return new Promise((resolve, reject) => {
    const u = new URL(url)
    const opts = { hostname: u.hostname, path: u.pathname + u.search, method: 'GET',
      rejectUnauthorized: false,   // verify manually below so Zscaler interception can be allowed
      headers: { 'User-Agent': 'LCL-updater', 'Accept': '*/*', ...headers } }
    const r = https.request(opts, resp => {
      // Trust gate: accept if Node already trusts the chain, or if the only
      // problem is an unknown issuer AND the presented chain is a Zscaler cert.
      const sock = resp.socket
      if (sock && sock.authorized === false) {
        const ae = String(sock.authorizationError || '')
        if (!(_ISSUER_ERRS.test(ae) && chainIsZscaler(sock))) {
          resp.resume(); sock.destroy()
          return reject(Object.assign(new Error('TLS certificate not trusted: ' + ae), { code: ae }))
        }
      }
      if (resp.statusCode >= 300 && resp.statusCode < 400 && resp.headers.location) {
        resp.resume()
        return resolve(httpsGetBuf(resp.headers.location, headers))
      }
      const chunks = []
      resp.on('data', c => chunks.push(c))
      resp.on('end', () => resolve({ statusCode: resp.statusCode, buf: Buffer.concat(chunks) }))
    })
    r.setTimeout(20000, () => r.destroy(new Error('update fetch timeout')))
    r.on('error', reject)
    r.end()
  })
}

async function fetchLatestRelease() {
  if (_updateCache.data && Date.now() - _updateCache.ts < 600000) return _updateCache.data
  const r = await httpsGetBuf('https://api.github.com/repos/' + UPDATE_REPO + '/releases/latest',
    { 'Accept': 'application/vnd.github+json' })
  if (r.statusCode !== 200) throw new Error('GitHub releases API returned ' + r.statusCode)
  const rel = JSON.parse(r.buf.toString('utf8'))
  const data = { tag: rel.tag_name, name: rel.name, notes: rel.body || '',
                 html_url: rel.html_url, published_at: rel.published_at }
  _updateCache = { ts: Date.now(), data }
  return data
}

function sha256Buf(buf) { return crypto.createHash('sha256').update(buf).digest('hex') }

async function handleUpdateApply(req, res) {
  try {
    const ch = readChannel()
    const ref = ch === 'alpha' ? ALPHA_REF : (await fetchLatestRelease()).tag
    if (!ref) return json(res, { error: 'No release tag found' }, 400)
    const csRes = await ghFetchFile(ref, 'checksums.txt')
    if (csRes.statusCode !== 200) {
      console.warn('[update] apply: checksums for ' + ref + ' HTTP ' + csRes.statusCode + (csRes.msg ? ' - ' + csRes.msg : ''))
      return json(res, { error: 'HTTP ' + csRes.statusCode }, 400)
    }
    const expected = parseChecksums(csRes.buf.toString('utf8'))
    // Download + verify ALL changed files first; apply nothing unless every one passes.
    const verified = {}, unchanged = []
    for (const fname of UPDATE_FILES) {
      const want = expected[fname]
      if (!want) throw new Error('No checksum published for ' + fname)
      const dest = path.join(LCL_DIR, fname)
      let cur = null
      try { cur = sha256Buf(fs.readFileSync(dest)) } catch {}
      if (cur === want) { unchanged.push(fname); continue }
      const dl = await ghFetchFile(ref, fname)
      if (dl.statusCode !== 200) throw new Error('Download failed for ' + fname + ' (HTTP ' + dl.statusCode + ')')
      if (sha256Buf(dl.buf) !== want) throw new Error('Checksum mismatch for ' + fname + ' - download rejected.')
      verified[fname] = dl.buf
    }
    // Integrity gate is the SHA-256 above; server.txt was already syntax-checked
    // by build.js before upload, so no redundant runtime check here.
    const applied = []
    for (const fname of Object.keys(verified)) {
      const dest = path.join(LCL_DIR, fname)
      const tmp = dest + '.tmp'
      fs.writeFileSync(tmp, verified[fname])
      fs.renameSync(tmp, dest)
      applied.push(fname)
    }
    _alphaCache = { ts: 0, data: null }
    json(res, {
      ok: true, channel: ch, ref, applied, unchanged,
      restartNeeded: applied.includes('server.txt'),
      refreshNeeded: applied.includes('index.html')
    })
  } catch (e) {
    json(res, { error: updateClientErr('apply', e) }, 500)
  }
}
// ---------------------------------------------------------------------------
// Update channels. 'stable' = GitHub releases (version-based, manual apply in
// the UI). 'alpha' = a tester channel tracking the 'alpha' branch BY FILE HASH,
// auto-applied at Node startup (both files together) before serving, re-exec'ing
// into the new server.txt. The choice is persisted server-side so Node knows it
// at boot. The SHA-256 checksum is the integrity gate; a re-exec guard prevents loops.
// ---------------------------------------------------------------------------
const ALPHA_REF = process.env.LCL_ALPHA_REF || 'alpha'
// Channel is stored in lcl_data.json (single data file) as appData.updateChannel.
function readChannel() {
  return (appData && appData.updateChannel === 'alpha') ? 'alpha' : 'stable'
}
function writeChannel(c) {
  if (!appData) return false
  appData.updateChannel = c === 'alpha' ? 'alpha' : 'stable'
  saveData(appData)
  _alphaCache = { ts: 0, data: null }
  return true
}
function parseChecksums(txt) {
  const m = {}
  for (const line of txt.split(/\r?\n/)) {
    const x = line.trim().match(/^([0-9a-fA-F]{64})\s+\*?(.+)$/)
    if (x) m[x[2].trim()] = x[1].toLowerCase()
  }
  return m
}
async function ghFetchFile(ref, name) {
  // Fetch a repo file as base64 JSON from api.github.com. NOT the raw media type
  // (it 302-redirects to raw.githubusercontent.com, which is 403-blocked on the
  // gov network), and always fresh (no CDN cache). Unauthenticated API is capped
  // at 60 req/hr per IP -> a terse 403 on the front end, full detail to console.
  const url = 'https://api.github.com/repos/' + UPDATE_REPO + '/contents/' + name + '?ref=' + encodeURIComponent(ref)
  const r = await httpsGetBuf(url, { 'Accept': 'application/vnd.github+json' })
  if (r.statusCode !== 200) {
    let msg = ''
    try { msg = JSON.parse(r.buf.toString('utf8')).message || '' } catch {}
    return { statusCode: r.statusCode, buf: Buffer.alloc(0), msg }
  }
  try {
    const j = JSON.parse(r.buf.toString('utf8'))
    if (j && j.encoding === 'base64' && typeof j.content === 'string') return { statusCode: 200, buf: Buffer.from(j.content, 'base64') }
    return { statusCode: 502, buf: Buffer.alloc(0) }
  } catch { return { statusCode: 502, buf: Buffer.alloc(0) } }
}
let _alphaCache = { ts: 0, data: null }
async function alphaStatus() {
  if (_alphaCache.data && Date.now() - _alphaCache.ts < 30000) return _alphaCache.data
  const cs = await ghFetchFile(ALPHA_REF, 'checksums.txt')
  if (cs.statusCode !== 200) throw new Error('@' + ALPHA_REF + ' checksums: HTTP ' + cs.statusCode + (cs.msg ? ' - ' + cs.msg : ''))
  const exp = parseChecksums(cs.buf.toString('utf8'))
  const changed = []
  for (const f of UPDATE_FILES) {
    const want = exp[f]; if (!want) continue
    let cur = null; try { cur = sha256Buf(fs.readFileSync(path.join(LCL_DIR, f))) } catch {}
    if (cur !== want) changed.push(f)
  }
  _alphaCache = { ts: Date.now(), data: { ref: ALPHA_REF, inSync: changed.length === 0, changed, hash: (exp['index.html'] || '').slice(0, 7) } }
  return _alphaCache.data
}
// All update log lines share one aligned label/column for a consistent console.
function ulog(m) { console.log('  ' + 'Updates'.padEnd(23) + '->  ' + m) }
function elog(m) { console.log('  ' + 'Embedding'.padEnd(23) + '->  ' + m) }
async function alphaBootstrap() {
  const cs = await ghFetchFile(ALPHA_REF, 'checksums.txt')
  if (cs.statusCode !== 200) { ulog('alpha: check failed - checksums HTTP ' + cs.statusCode + (cs.msg ? ' (' + cs.msg + ')' : '')); return }
  const exp = parseChecksums(cs.buf.toString('utf8'))
  // Per-file hash compare. Stay quiet when up to date — only files that actually
  // differ are logged (below), so the routine "in sync" case is a single line.
  const changed = []
  for (const f of UPDATE_FILES) {
    if (!exp[f]) { ulog('alpha: no checksum for ' + f + ' - skipping'); return }
    let cur = null; try { cur = sha256Buf(fs.readFileSync(path.join(LCL_DIR, f))) } catch {}
    if (cur !== exp[f]) changed.push(f)
  }
  if (!changed.length) { ulog('alpha: in sync (v' + APP_VERSION + ')'); return }
  ulog('alpha: updating ' + changed.join(', '))
  backupStable()   // snapshot current stable files before overwriting (offline fallback)
  // Download + verify ONLY the changed file(s) (the SHA-256 is the integrity gate;
  // build.js already syntax-checked before upload).
  const bufs = {}
  for (const f of changed) {
    const dl = await ghFetchFile(ALPHA_REF, f)
    if (dl.statusCode !== 200) throw new Error('download ' + f + ' HTTP ' + dl.statusCode)
    if (sha256Buf(dl.buf) !== exp[f]) throw new Error('checksum mismatch ' + f)
    bufs[f] = dl.buf
    ulog('downloaded ' + f.padEnd(11) + exp[f].slice(0, 7))
  }
  for (const f of changed) {
    const dest = path.join(LCL_DIR, f), tmp = dest + '.tmp'
    fs.writeFileSync(tmp, bufs[f]); fs.renameSync(tmp, dest)
  }
  ulog('applied     ' + changed.join(', '))
  // Default: do NOT auto re-exec. A new server.txt can only load in a fresh Node
  // process, and auto-spawning one freezes the parent REPL behind spawnSync and
  // ends the session. Instead, apply the file and ask the user to restart — the
  // REPL stays alive. (The explicit front-end 'Update & restart' button still
  // restarts on demand via handleUpdateRestart.)
  if (changed.includes('server.txt')) ulog('server.txt updated - restart Node to load the new server')
  if (changed.includes('index.html')) ulog('index.html updated - hard-refresh the browser (no restart needed)')
}
async function restoreStable() {
  const rel = await fetchLatestRelease()
  const tag = rel.tag; if (!tag) throw new Error('no release tag')
  const cs = await ghFetchFile(tag, 'checksums.txt')
  if (cs.statusCode !== 200) throw new Error('no checksums in ' + tag)
  const exp = parseChecksums(cs.buf.toString('utf8'))
  const bufs = {}
  for (const f of UPDATE_FILES) {
    const want = exp[f]; if (!want) throw new Error('no checksum for ' + f)
    const dl = await ghFetchFile(tag, f)
    if (dl.statusCode !== 200) throw new Error('download ' + f + ' HTTP ' + dl.statusCode)
    if (sha256Buf(dl.buf) !== want) throw new Error('checksum mismatch ' + f)
    bufs[f] = dl.buf
  }
  const applied = []
  for (const f of UPDATE_FILES) {
    const dest = path.join(LCL_DIR, f), tmp = dest + '.tmp'
    fs.writeFileSync(tmp, bufs[f]); fs.renameSync(tmp, dest); applied.push(f)
  }
  return { tag, applied }
}
// .stable backup helpers — keep a verbatim copy of the stable files so a revert
// can fall back to them when GitHub is unreachable.
function stableBackupPath(f) { return path.join(LCL_DIR, f + '.stable') }
function backupStable() {
  for (const f of UPDATE_FILES) {
    const src = path.join(LCL_DIR, f), bak = stableBackupPath(f)
    try { if (fs.existsSync(src) && !fs.existsSync(bak)) fs.copyFileSync(src, bak) }
    catch (e) { ulog('backup ' + f + ' failed: ' + (e.code || e.message)) }
  }
}
function restoreFromBackup() {
  const restored = []
  for (const f of UPDATE_FILES) {
    const dest = path.join(LCL_DIR, f), bak = stableBackupPath(f)
    try { if (fs.existsSync(bak)) { const tmp = dest + '.tmp'; fs.copyFileSync(bak, tmp); fs.renameSync(tmp, dest); restored.push(f) } }
    catch (e) { ulog('restore ' + f + ' failed: ' + (e.code || e.message)) }
  }
  return restored
}
function clearStableBackup() { for (const f of UPDATE_FILES) { try { fs.unlinkSync(stableBackupPath(f)) } catch {} } }

// Download + verify + atomically swap UPDATE_FILES for a given git ref. The
// SHA-256 from that ref's checksums.txt is the integrity gate. Returns
// { applied, unchanged }; throws (applying nothing) if any file fails.
async function applyRef(ref) {
  const cs = await ghFetchFile(ref, 'checksums.txt')
  if (cs.statusCode !== 200) throw new Error('checksums HTTP ' + cs.statusCode + (cs.msg ? ' - ' + cs.msg : ''))
  const exp = parseChecksums(cs.buf.toString('utf8'))
  const verified = {}, unchanged = []
  for (const f of UPDATE_FILES) {
    const want = exp[f]; if (!want) throw new Error('no checksum for ' + f)
    let cur = null; try { cur = sha256Buf(fs.readFileSync(path.join(LCL_DIR, f))) } catch {}
    if (cur === want) { unchanged.push(f); continue }
    const dl = await ghFetchFile(ref, f)
    if (dl.statusCode !== 200) throw new Error('download ' + f + ' HTTP ' + dl.statusCode)
    if (sha256Buf(dl.buf) !== want) throw new Error('checksum mismatch ' + f)
    verified[f] = dl.buf
  }
  const applied = []
  for (const f of Object.keys(verified)) {
    const dest = path.join(LCL_DIR, f), tmp = dest + '.tmp'
    fs.writeFileSync(tmp, verified[f]); fs.renameSync(tmp, dest); applied.push(f)
  }
  _alphaCache = { ts: 0, data: null }
  return { applied, unchanged }
}

function handleUpdateChannel(req, res) {
  readBody(req, (err, body) => {
    if (err) return json(res, { error: 'bad request body' }, 400)
    const ch = (body && body.channel === 'alpha') ? 'alpha' : 'stable'

    if (ch === 'alpha') {
      // Snapshot stable (offline fallback), then download + swap the latest alpha
      // build. On any failure, apply nothing and stay on stable.
      backupStable()
      applyRef(ALPHA_REF).then(r => {
        writeChannel('alpha')
        json(res, { ok: true, channel: 'alpha', applied: r.applied, unchanged: r.unchanged,
                    restartNeeded: r.applied.includes('server.txt'),
                    refreshNeeded: r.applied.includes('index.html') })
      }).catch(e => {
        clearStableBackup()   // nothing was applied; drop the just-made backup
        json(res, { error: friendlyUpdateErr(e), channel: 'stable' }, 500)
      })
      return
    }

    // Revert: pull the latest stable release from GitHub; if that fails (offline),
    // fall back to the .stable backup. Drop the backup once back on stable.
    writeChannel('stable')
    restoreStable().then(r => {
      clearStableBackup()
      json(res, { ok: true, channel: 'stable', restored: r, restartNeeded: true })
    }).catch(e => {
      const fb = restoreFromBackup()
      if (fb.length) { clearStableBackup(); json(res, { ok: true, channel: 'stable', restoredFromBackup: fb, restartNeeded: true }) }
      else json(res, { ok: true, channel: 'stable', restoreError: friendlyUpdateErr(e) })
    })
  })
}

function handleUpdateRestart(req, res) {
  json(res, { ok: true, restarting: true })
  // Flush the response, then release the port and re-exec into the (already-applied)
  // server.txt in the same terminal. The browser polls /api/health and reloads.
  setTimeout(() => {
    try {
      ulog('front-end requested restart - re-exec into server.txt ...')
      try { server.close() } catch {}
      for (const sock of openSockets) { try { sock.destroy() } catch {} }
      const r = require('child_process').spawnSync(process.execPath, [__filename], { stdio: 'inherit', env: Object.assign({}, process.env, { LCL_ALPHA_BOOTED: '1' }) })
      if (r.error) { ulog('restart spawn failed (' + r.error.code + ') - restart Node manually'); process.exit(1) }
      process.exit(r.status == null ? 0 : r.status)
    } catch (e) {
      ulog('restart failed (' + (e.code || e.message) + ') - restart Node manually')
    }
  }, 350)
}

const server = http.createServer((req, res) => {
  // Host-header guard: refuse anything that wasn't addressed to localhost.
  if (!isLocalHost(req)) {
    res.writeHead(403, { 'Content-Type': 'text/plain; charset=utf-8' })
    res.end('Forbidden: localhost only')
    return
  }

  // Reflect the loopback origin only; no wildcard CORS.
  const origin = req.headers.origin
  if (origin && /^https?:\/\/(localhost|127\.0\.0\.1|\[::1\])(:\d+)?$/i.test(origin)) {
    res.setHeader('Access-Control-Allow-Origin', origin)
    res.setHeader('Vary', 'Origin')
  }
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS')
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type')

  if (req.method === 'OPTIONS') {
    res.writeHead(204)
    res.end()
    return
  }

  const p = req.url.split('?')[0]

  if (req.method === 'POST' && p === '/api/chat')          return handleChat(req, res)
  if (req.method === 'POST' && p === '/api/embed')         return handleEmbed(req, res)
  if (req.method === 'POST' && p === '/api/embed-batch')   return handleEmbedBatch(req, res)
  if (req.method === 'POST' && p === '/api/embed-lookup')  return handleEmbedLookup(req, res)
  if (req.method === 'POST' && p === '/api/embed-evict')   return handleEmbedEvict(req, res)
  if (req.method === 'POST' && p === '/api/embed-gc')      return handleEmbedGc(req, res)
  if (req.method === 'GET'  && p === '/api/config')       return handleGetConfig(req, res)
  if (req.method === 'POST' && p === '/api/config')       return handleSaveConfig(req, res)
  if (req.method === 'GET'  && p === '/api/data')         return handleGetData(req, res)
  if (req.method === 'POST' && p === '/api/data')         return handleSaveData(req, res)

  if (req.method === 'GET'  && p === '/api/update/check') return handleUpdateCheck(req, res)
  if (req.method === 'POST' && p === '/api/update/apply') return handleUpdateApply(req, res)
  if (req.method === 'POST' && p === '/api/update/channel') return handleUpdateChannel(req, res)
  if (req.method === 'POST' && p === '/api/update/restart') return handleUpdateRestart(req, res)
  if (req.method === 'GET'  && p === '/api/health')        return json(res, { ok: true, version: APP_VERSION })

  if (req.method === 'GET'  && p === '/skills') return handleSkillsList(req, res)
  if (req.method === 'POST' && p === '/skills/reload') return handleSkillsList(req, res)

  // /skills/:id  (no further slashes allowed)
  const skillMatch = /^\/skills\/([^\/]+)$/.exec(p)
  if (skillMatch) {
    const id = decodeURIComponent(skillMatch[1])
    if (req.method === 'GET')    return handleSkillRead(req, res, id)
    if (req.method === 'PUT')    return handleSkillWrite(req, res, id)
    if (req.method === 'DELETE') return handleSkillDelete(req, res, id)
  }

  return serveStatic(req, res)
})

// Track live sockets so a restart can free the port promptly.
const openSockets = new Set()
server.on('connection', sock => { openSockets.add(sock); sock.on('close', () => openSockets.delete(sock)) })

// Graceful shutdown — make Ctrl+C (SIGINT) and SIGTERM reliably stop the server.
// Critical when launched from inside the Node REPL (`require('./server.txt')`),
// whose own SIGINT handler otherwise swallows Ctrl+C, and under the alpha
// re-exec supervisor. A 2nd Ctrl+C (or the 300ms timer) forces exit if a
// keep-alive socket or other handle lingers.
let _shuttingDown = false
function shutdown(sig) {
  if (_shuttingDown) { process.exit(0); return }
  _shuttingDown = true
  try { console.log('\n  Shutting down (' + sig + ') - bye.') } catch {}
  try { server.close() } catch {}
  for (const sock of openSockets) { try { sock.destroy() } catch {} }
  try { upstreamAgent.destroy() } catch {}
  setTimeout(() => process.exit(0), 300).unref()
}
process.on('SIGINT', () => shutdown('SIGINT'))
process.on('SIGTERM', () => shutdown('SIGTERM'))
// The alpha auto-restart child is spawned (spawnSync, inherited stdio) from the
// parent REPL, so it inherits the REPL's raw-mode TTY. In raw mode Ctrl+C is a
// keystroke, not a SIGINT, and the child isn't a REPL — so it never stops, while
// the parent is blocked in spawnSync (which ignores SIGINT). Restore cooked mode
// so Ctrl+C raises a real SIGINT (handled above); the child exiting unblocks the
// parent. A keypress watch is kept as a fallback. Scoped to the spawned child
// (LCL_ALPHA_BOOTED) so the parent REPL's own input handling is left untouched.
if (process.env.LCL_ALPHA_BOOTED === '1') {
  try {
    const stdin = process.stdin
    if (stdin && stdin.isTTY) {
      try { if (stdin.isRaw) stdin.setRawMode(false) } catch {}
      require('readline').emitKeypressEvents(stdin)
      stdin.on('keypress', (ch, key) => {
        if (key && key.ctrl && (key.name === 'c' || key.name === 'd')) shutdown('Ctrl+' + key.name.toUpperCase())
      })
    }
  } catch {}
}

async function boot() {
  // Print the startup banner first, then run the alpha check so its lines appear
  // beneath the banner rather than scrolling above it.
  server.listen(PORT, BIND_HOST, () => {
    startupLog()
    if (readChannel() === 'alpha') {
      alphaBootstrap().catch(e => ulog('alpha: check skipped (' + e.message + ')'))
    }
  })
}
boot()
